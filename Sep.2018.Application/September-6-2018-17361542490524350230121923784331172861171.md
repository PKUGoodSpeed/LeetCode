            }

                    }
                }

        public boolean isValid(String s) {
            int count = 0;

                    count++;
                } else if (s.charAt(i) == ')') {
                    count--;
                }
                if (count < 0) return false;
            }
            return count == 0;
        }

        public ListNode rotateRight(ListNode head, int k) {
            if (k == 0 || head == null || head.next == null) return head;

            ListNode node = head;
